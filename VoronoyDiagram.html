# Importamos las librerias necesarias para el algoritmo Voronoi
import numpy as np
from scipy.spatial import Voronoi, voronoi_plot_2d
import matplotlib.pyplot as plt

# Necesitamos un array con la latitud y la longitud
puntos = data_spain[['Longitude', 'Latitude']].values

# Creamos el objeto Voronoi
voronoi = Voronoi(puntos)

# Dibujamos el diagrama
fig = voronoi_plot_2d(voronoi, show_vertices=False)
for region in voronoi.regions:
    if not -1 in region:
        polygon = [voronoi.vertices[i] for i in region]
        plt.fill(*zip(*polygon), alpha=0.5)
plt.show()

Vemos algunas regiones en blanco porque son infinitas, es decir, en el diagrama anterior no están definidas todas sus fronteras.
