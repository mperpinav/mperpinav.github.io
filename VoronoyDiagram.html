# Importamos las librerias necesarias para el algoritmo Voronoi
import numpy as np
from scipy.spatial import Voronoi, voronoi_plot_2d
import matplotlib.pyplot as plt

# Necesitamos un array con la latitud y la longitud y los nombres de las capitales
puntos = data_spain[['Longitude', 'Latitude']].values
capitales = data_spain['AccentCity'].values
# Creamos el objeto Voronoi
voronoi = Voronoi(puntos)

# Dibujamos el diagrama
fig, axs = plt.subplots(1, figsize=(12, 6), dpi=300)
plt.xlim(0, 12)
plt.ylim(0, 8)
plt.axis('off')
axs.set_aspect('equal')
voronoi_plot_2d(voronoi, show_vertices=False, ax=axs)
for region in voronoi.regions:
    if not -1 in region:
        polygon = [voronoi.vertices[i] for i in region]
        plt.fill(*zip(*polygon), alpha = 0.6)

# Añadimos el nombre de las capitales correspondiente a cada semilla
for i, point in enumerate(puntos):
    plt.text(point[0], point[1], capitales[i], fontsize=7, ha='center', color='black')

plt.show()
